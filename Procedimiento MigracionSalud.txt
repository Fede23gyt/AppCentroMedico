CREATE procedure [dbo].[MigracionSalud]

@iAnioPeriodo INT,@iMesPeriodo INT,@iIdTitularCp INT,@bBeneficiarioCobertura int,@bInformacionPago int,@bDeterminacionCarencia int,@iCarencia INT,@iIdSucursal int

AS 

--USE seodesarrollo2

-- 2018,10,-1,1,1,1,2,1
/*
SET @iAnioPeriodo =2018
SET @iMesPeriodo =10
SET @iIdTitularCp =-1
SET @bBeneficiarioCobertura =1
SET @bInformacionPago =1
SET @bDeterminacionCarencia =1
SET @iCarencia =2
SET @iIdSucursal =1
*/

/*
	Procdimiento almacenado para la migracion de los datos desde sql salud al sistema de emision de ordenes
	Fecha 23-03-2013
	Autor CU LGP
*/

--SALSR1 --->PIE
--PIEANTA  ---> PJV
--PIEVEANTA --> PJV

declare @FechaSinVigenciaIncorporacion datetime
DECLARE @iIdError INT

SET @iIdError=0	--

set @FechaSinVigenciaIncorporacion={ d '2070-01-01'} --Sin Vigencia hasta la incorporacion del primer pago en empresa

-- Determinacion del periodo que corresponde
set @iAnioPeriodo=year(getdate())
set @iMesPeriodo=month(getdate())

if (day(getdate()))<=10
begin
	if month(getdate())=1 
	begin
		set @iAnioPeriodo=year(getdate()) -1
		set @iMesPeriodo=12
	end
	begin		
		set @iMesPeriodo= month(getdate())-1
	end

end

print 'Migrando a SEo Salud'
print @iAnioPeriodo
print @iMesPeriodo


IF @bBeneficiarioCobertura=1
begin
	PRINT 'Actualizando Personas y coberturas'

	--BEGIN TRAN Migracion_Afiliados --Proceso transaccional

	--Eliminacion de todas las coberturas.
	PRINT 'Eliminaci n de todas las coberturas'
	exec gpievenew.seodesarrollo2.dbo.ups_BorrarTablaCoberturaBeneficiario @iIdTitularCp
	--DELETE cb 
	--from gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios cb		   
	--inner Join gpievenew.seodesarrollo2.dbo.Beneficiarios Beneficiarios 
	--on Beneficiarios.IdBenCp=cb.IdBenCf
	--INNER JOIN gpievenew.seodesarrollo2.dbo.Fichas f ON f.IdConvenioCf=1 AND f.IdTitularCp=Beneficiarios.IdTitularCf
	--INNER JOIN gpieve.PieveSalud.dbo.fichas FichaPs ON f.idtitularcp=FichaPs.IdTitularCp
	--where 
	--(Beneficiarios.IdTitularCf= @iIdTitularCp OR @iIdTitularCp=-1)
	--and f.idConvenioCf=1	-- solo eliminanos el convenio de Pieve Salud
	SET @iIdError=@@ERROR

	
	--select idconveniocf,count(*) from gpievenew.seodesarrollo2.dbo.Fichas group by idconveniocf
	
	
	print 'Eliminacion de todas las coberturas(Incluidas la de los sistemas satelites.)'
	--Delete cb 
	--from gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios cb
	--inner Join gpievenew.seodesarrollo2.dbo.Beneficiarios Beneficiarios 
	--on Beneficiarios.IdBenCp=cb.IdBenCf
	---- fichas solo considerar la del convenio numero 1 (de pieve salud)
	--INNER JOIN gpievenew.seodesarrollo2.dbo.Fichas f ON f.IdConvenioCf=1 AND f.IdTitularCp=Beneficiarios.IdTitularCf


	--where 
	--(Beneficiarios.IdTitularCf= @iIdTitularCp OR @iIdTitularCp=-1)
	--AND beneficiarios.IdTitularCf<>999999
	--SET @iIdError=@@ERROR
	
	print 'Eliminacion de los beneficiarios de los sitemas satelites (aquellos que no estan en pievesalud.)'
	DELETE ben
	FROM gpievenew.seodesarrollo2.dbo.Beneficiarios ben
	INNER JOIN gpievenew.seodesarrollo2.dbo.Fichas f ON f.IdConvenioCf=1 AND f.IdTitularCp=ben.IdTitularCf -- eliminacion solo de convenio de Pieve salud
	LEFT JOIN gpieve.PieveSalud.dbo.Fichas FichPS ON fichPS.IdTitularCp=ben.IdTitularCf	
	WHERE fichps.IdTitularCp IS NULL
	AND (ben.IdTitularCf=@iIdTitularCp OR @iIdTitularCp=-1)
	AND ben.IdTitularCf<>999999
	
	print 'Eliminacion de las ficha de los sistemas (aquellas que no estan en pieve salud)'
	DELETE fic
	FROM gpievenew.seodesarrollo2.dbo.Fichas fic 
	LEFT JOIN gpieve.PieveSalud.dbo.Fichas FichPS ON fichPS.IdTitularCp=fic.IdTitularCp
	WHERE fichps.IdTitularCp IS NULL
	AND (fic.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1) AND fic.IdTitularCp<>999999
	and fic.idConvenioCf=1 -- no tocams las coberturas particulares.
	
	
	--Marcar de baja todos los beneficiarios estan de baja en el sistema de pieve salud.
	PRINT 'Marcar de baja todos los beneficiarios estan de baja en el sistema de pieve salud.'
	IF @iIdError=0
	BEGIN
		
		Update benSeo set benSeo.Existe =0 
		from gpievenew.seodesarrollo2.dbo.Beneficiarios benSeo 
		inner Join 
		gpieve.PieveSalud.dbo.beneficiarios Ben on Ben.IdBenCp=benSeo.IdBenCp and 
		Ben.Existe=0 --Aquellas personas de baja en pieve salud.
		WHERE (benseo.IdTitularCf=@iIdTitularCp OR @iIdTitularCp=-1)
		
		SET @iIdError=@@ERROR	
	END

	--Marcar de baja todas las fichas que estan de baja en el sistema de pieve salud.
	PRINT 'Marcar de baja todas las fichas que estan de baja en el sistema de pieve salud'
	IF @iIdError=0
	BEGIN	
		Update FichasSeo 
		set FichasSeo.Existe=0 
		from seodesarrollo2.dbo.Fichas FichasSeo 
		inner Join gpieve.PieveSalud.dbo.Fichas Fic on Fic.IdTitularCp=FichasSeo.IdTitularcp
		and fic.Existe=0  --Fichas dadas de baja en el sistema de pieve salud.
		WHERE 
		(FichasSeo.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		
		SET @iIdError=@@ERROR                                      
	END


	--Actualizar las fichas existentes en ambos lados
	PRINT 'Actualizar las fichas existentes en ambos lados'
	IF @iIdError=0
	BEGIN
		Update FicsSeo
		SET FicsSeo.Existe=1,
			FicsSeo.Telefono=substring(Telefonos,1,20) ,
			FicsSeo.Domicilio=fichps.Direccion + ',' + fichps.BarrioReal + ', ' + isnull(nomlocalidad,'')
		FROM gpievenew.seodesarrollo2.dbo.Fichas FicsSeo
		INNER JOIN
		gpieve.PieveSalud.dbo.Fichas FichPS	ON fichps.IdTitularCp=ficsseo.IdTitularCp	
		LEFT JOIN gpieve.PieveSalud.dbo.Localidades l ON l.IdLocalidadCp=FichPS.IdLocalidadRCF
		WHERE 
		FichPS.Existe=1 AND 
		fichps.IdTitularCp>0
		AND (FichPS.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		
		SET @iIdError=@@ERROR

	END

	--Insertar las fichas inexistentes en seo.
	PRINT 'Insertar las fichas inexistentes en seo.'
	IF @iIdError=0
	BEGIN
		INSERT INTO gpievenew.seodesarrollo2.dbo.Fichas(IdTitularCp,Existe, IdConvenioCf,Telefono,Domicilio,fechaVigenciaPago)
		SELECT fichps.IdTitularCp,fichps.Existe,1 idConvencioCf,substring(Telefonos,1,20) sTelefono,
		fichps.Direccion + ',' + fichps.BarrioReal + ', ' + isnull(nomlocalidad,'')	 ,@FechaSinVigenciaIncorporacion
		FROM 
		gpieve.PieveSalud.dbo.Fichas FichPS
		LEFT JOIN gpieve.PieveSalud.dbo.Localidades l ON l.IdLocalidadCp=FichPS.IdLocalidadRCF
		--QUE NO ESTAN.
		LEFT JOIN gpievenew.seodesarrollo2.dbo.Fichas Ficseo ON FicSeo.IdTitularCp=FichPs.IdTitularCp
		WHERE 
		FichPS.Existe=1 AND fichps.IdTitularCp>0
		AND (FichPS.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		AND Ficseo.IdTitularCp IS NULL	--No deben estar en el sistema SEO.
		
		SET @iIdError=@@ERROR
	END

	--Actualizar datos de beneficiarios existentes en ambos lados.
	PRINT 'Actualizar datos de beneficiarios existentes en ambos lados'
	print 'error'
	print @iIderror
	if @iIdError=0
	begin
	print 'entro a la actualizacion de personas'
		Update benSeo 
		set benSeo.Existe =1,
			benSeo.Apellido = ben.Apellido,
			benseo.Nombre = ben.Nombre,
			benseo.TDoc = substring(ben.DocTipo,1, 3),
			benseo.NumDoc = substring(ben.dni,1, 10),
			benseo.FechNac = case when isnull(ben.FechNac,GETDATE()) <= { d '1900-01-01'} or isnull(ben.FechNac,GETDATE()) >= { d '2020-01-01'} THEN GETDATE() ELSE ben.FechNac END,
			benseo.IdVinculoCf = ben.IdVinculoCf
		 
		from gpievenew.seodesarrollo2.dbo.Beneficiarios benSeo 
		inner Join 
		gpieve.PieveSalud.dbo.beneficiarios Ben on Ben.IdBenCp=benSeo.IdBenCp
		--Sobre las fichas activas.    
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=ben.IdTitularCF AND f.IdTitularCp>0 AND f.Existe=1
		WHERE (f.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		and Ben.Existe=1 --personas activas en pieve salud.
		
		SET @iIdError=@@ERROR	
	    
	END

	--Insertar los beneficiarios inexistentes en seo.
	PRINT 'Insertar los beneficiarios inexistentes en seo'
	if @iIdError=0
	BEGIN

		INSERT INTO gpievenew.seodesarrollo2.dbo.Beneficiarios(IdTitularCf,IdBenCp,Parent,FechaAlta,Existe,Apellido,
													  Nombre,TDoc,NumDoc,FechNac,IdVinculoCf)
		SELECT ben.IdTitularCF,ben.IdBenCP,isnull(ben.Parent,0),
		CASE WHEN ISNULL(ben.FechaIng,GETDATE()) <= { d '1900-01-01' } OR ISNULL(ben.FechaIng,GETDATE()) > GETDATE() THEN GETDATE() ELSE ISNULL(ben.FechaIng,GETDATE())END,
		1 existe,
		ben.Apellido,
		ben.Nombre,
		substring(ben.DocTipo,1, 3),
		substring(ben.dni,1, 10),
		case when isnull(ben.FechNac,GETDATE()) <= { d '1900-01-01'} THEN GETDATE() ELSE case when ben.FechNac>GETDATE() THEN getdate() ELSE ben.FechNac end END,
		ben.IdVinculoCf	
		from  	 
		gpieve.PieveSalud.dbo.beneficiarios Ben 
		--Sobre las fichas activas.    
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=ben.IdTitularCF AND f.IdTitularCp>0 AND f.Existe=1
		LEFT JOIN gpievenew.seodesarrollo2.dbo.Beneficiarios benSeo ON benseo.IdBenCp=ben.IdBenCP    
		WHERE (f.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		and Ben.Existe=1 --personas activas en pieve salud.
		AND benseo.IdBenCp IS NULL	--No se encuentra en el sistema de emision de ordenes.
		
		SET @iIdError=@@ERROR	

	end

	-- Ahora transferencia de coberturas.

	--Cobertura sobre Odontologia
	PRINT 'Cobertura sobre Odontologia'
	if @iIdError=0
	begin	 
		--INSERT INTO gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios(IdBenCf,CodCoberturaCf,FechaAlta)
			 
		SELECT distinct beneficiarios.IdBenCP,ISNULL(SUBSTRING(co.Cobertura,1,3),'ODO') CodCoberturaCf,	
		case when isnull(PlanesBeneficiarios.FechaAlta,GETDATE())<={ d '1900-01-01'} OR
				  isnull(PlanesBeneficiarios.FechaAlta,GETDATE())>{ d '2019-01-01'} THEN 
	          		GETDATE() ELSE isnull(PlanesBeneficiarios.FechaAlta,GETDATE()) END FechaAlta  
		INTO #tmp_CoberturaOdontologia
		FROM  gpieve.PieveSalud.dbo.PlanesFichas PlanesFichas 
		inner Join gpieve.PieveSalud.dbo.Beneficiarios Beneficiarios on PlanesFichas.IdTitularCf=Beneficiarios.IdTitularCf
		
		AND planesfichas.IdRubroCf=3		--Odontologia                   
		Inner Join gpieve.PieveSalud.dbo.PlanesBeneficiarios PlanesBeneficiarios on PlanesBeneficiarios.IdRubroCf=PlanesFichas.idRubroCf and
		Beneficiarios.IdBenCp=PlanesBeneficiarios.IdBenCf and Beneficiarios.Existe=1
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=PlanesFichas.IdTitularCF AND f.Existe=1 AND f.IdTitularCp>0
		left JOIN gpieve.PieveSalud.dbo.CoberturaOdontologia co ON co.IdRubroCf=3 AND co.PL=SUBSTRING(PlanesFichas.[Plan],1,3)
		WHERE (f.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		
		SET @iIdError=@@ERROR	
	END

	--SELECT * FROM CoberturaOdontologia

	--Cobertura sobre Odontologia no encontrado
	PRINT 'Cobertura sobre Odontologia no encontrado'
	if @iIdError=0
	begin	 
		--INSERT INTO gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios(IdBenCf,CodCoberturaCf,FechaAlta)
			 
		SELECT beneficiarios.IdBenCP,
		max(SUBSTRING(co.Cobertura,1,3)) CodCoberturaCf,	
		MIN(case when isnull(PlanesBeneficiarios.FechaAlta,GETDATE())<={ d '1900-01-01'} OR
				  isnull(PlanesBeneficiarios.FechaAlta,GETDATE())>{ d '2019-01-01'} THEN 
				  GETDATE() ELSE isnull(PlanesBeneficiarios.FechaAlta,GETDATE()) END) FechaAlta  
		into #tmp_CoberturaSinOdontologia 	          
		FROM  gpieve.PieveSalud.dbo.PlanesFichas PlanesFichas 
		inner Join gpieve.PieveSalud.dbo.Beneficiarios Beneficiarios on PlanesFichas.IdTitularCf=Beneficiarios.IdTitularCf	
		AND planesfichas.IdRubroCf<>3		--Odontologia                   	
		Inner Join gpieve.PieveSalud.dbo.PlanesBeneficiarios PlanesBeneficiarios on PlanesBeneficiarios.IdRubroCf=PlanesFichas.idRubroCf and
		Beneficiarios.IdBenCp=PlanesBeneficiarios.IdBenCf and Beneficiarios.Existe=1 
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=PlanesFichas.IdTitularCF AND f.Existe=1 AND f.IdTitularCp>0
		
		inner JOIN gpieve.PieveSalud.dbo.CoberturaOdontologia co ON			--Debe encontrarse.
		co.IdRubroCf=0 AND co.PL=SUBSTRING(PlanesFichas.[Plan],1,3)
		 
		-- que no esta en las coberturas. 	
		LEFT JOIN #tmp_CoberturaOdontologia cb ON cb.IdBenCp=beneficiarios.IdBenCP 	
		WHERE (f.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)	and
		 cb.IdBenCp IS NULL				--No insertado en la de odontologia.
		GROUP BY beneficiarios.IdBenCP
		
			
		--Detectar posible error
		SET @iIdError=@@ERROR
	END

	--Si no esta en ni en el primero ni en el segundo entonces insertamos simplemente "ODO"
	PRINT 'Si no esta en ni en el primero ni en el segundo entonces insertamos simplemente "ODO"'
	if @iIdError=0
	BEGIN
		SELECT distinct beneficiarios.IdBenCP,
		'ODO' CodCoberturaCf,
		CONVERT(DATETIME, 
		CONVERT(VARCHAR, isnull(f.AnioPeriodoIngreso,2010)) + '-' + convert(VARCHAR,ISNULL(f.MesPeriodoIngreso,1)) + '-01' 
		)		
		 FechaAlta  
		into #tmp_CoberturaOdoFinal 	          
		FROM  gpieve.PieveSalud.dbo.PlanesFichas PlanesFichas 
		inner Join gpieve.PieveSalud.dbo.Beneficiarios Beneficiarios on PlanesFichas.IdTitularCf=Beneficiarios.IdTitularCf	
		Inner Join gpieve.PieveSalud.dbo.PlanesBeneficiarios PlanesBeneficiarios on PlanesBeneficiarios.IdRubroCf=PlanesFichas.idRubroCf and
		Beneficiarios.IdBenCp=PlanesBeneficiarios.IdBenCf and Beneficiarios.Existe=1 
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=PlanesFichas.IdTitularCF AND f.Existe=1 AND f.IdTitularCp>0
		
		-- que no esta en las coberturas. 	
		LEFT JOIN #tmp_CoberturaOdontologia cb ON cb.IdBenCp=beneficiarios.IdBenCP
		LEFT JOIN #tmp_CoberturaSinOdontologia cbs ON cbs.IdBenCp=beneficiarios.IdBenCP 	
		WHERE (f.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)	and
		 cb.IdBenCp IS NULL and				--No insertado en la de odontologia.
		 cbs.idBenCp IS NULL				--No es ODO 	
		and AnioPeriodoIngreso >=1900 and MesPeriodoIngreso between 1 and 12
		--Detectar posible error
		SET @iIdError=@@ERROR
	END

	--INSERTAR LAS TRES OPCIONES ANTERIORES.
	PRINT 'Insertando todo'
	if @iIdError=0
	BEGIN
		INSERT INTO gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios(IdBenCf,CodCoberturaCf,FechaAlta)
		
		SELECT distinct t.idBenCp,t.CodCoberturaCf,t.FechaAlta	 
		FROM 
		(SELECT idbencp,CodCoberturaCf COLLATE Compatibility_54_c0a_30003 CodCoberturaCf , FechaAlta  FROM #tmp_CoberturaOdontologia
		UNION all
		select  idbencp,CodCoberturaCf COLLATE Compatibility_54_c0a_30003 CodCoberturaCf , FechaAlta  from #tmp_CoberturaSinOdontologia
		UNION ALL 
		select  idbencp,CodCoberturaCf COLLATE Compatibility_54_c0a_30003 CodCoberturaCf , FechaAlta  from #tmp_CoberturaOdoFinal) t
		
		SET @iIdError=@@ERROR
		
	END



	--INSERTAR Los Sin SAL (todo el mundo)
	PRINT 'INSERTAR Los Sin SAL (todo el mundo)'
	if @iIdError=0
	BEGIN
		INSERT INTO gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios(IdBenCf,CodCoberturaCf,FechaAlta)
		
		SELECT distinct t.idBenCp,'SINSAL' CodCoberturaCf,		
		isnull(
			planesbeneficiarios.FechaAlta,
		CONVERT(DATETIME, 
		CONVERT(VARCHAR, isnull(f.AnioPeriodoIngreso,2010)) + '-' + convert(VARCHAR,CASE WHEN ISNULL(f.MesPeriodoIngreso,1)<1 THEN 1 ELSE ISNULL(f.MesPeriodoIngreso,1) END ) + '-01' 
		)
		) FechaAlta
		 
		FROM 
		(SELECT  IdBenCp , CodCoberturaCf  COLLATE Compatibility_54_c0a_30003 CodCoberturaCf,FechaAlta    FROM #tmp_CoberturaOdontologia
		UNION all
		select  IdBenCp , CodCoberturaCf  COLLATE Compatibility_54_c0a_30003 CodCoberturaCf,FechaAlta  from #tmp_CoberturaSinOdontologia
		UNION ALL 
		select  IdBenCp , CodCoberturaCf  COLLATE Compatibility_54_c0a_30003 CodCoberturaCf,FechaAlta   from #tmp_CoberturaOdoFinal) 
		t
		INNER JOIN gpieve.PieveSalud.dbo.Beneficiarios b ON b.IdBenCP=t.IdBenCP
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=b.IdTitularCF AND  f.AnioPeriodoIngreso<=YEAR(GETDATE())+1
		AND f.AnioPeriodoIngreso >=1980
		left join gpieve.PieveSalud.dbo.PlanesFichas PlanesFichas ON planesfichas.IdTitularCF=b.IdTitularCF AND planesfichas.IdRubroCf=1
		left JOIN gpieve.PieveSalud.dbo.PlanesBeneficiarios PlanesBeneficiarios on PlanesBeneficiarios.IdRubroCf=1 and
		PlanesBeneficiarios.IdBenCf=t.IdBenCP
		
		SET @iIdError=@@ERROR
		
	END


	-- Rubro SALUD
	PRINT 'Rubro SALUD'
	if @iIdError=0
	BEGIN		
		INSERT INTO gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios(IdBenCf,CodCoberturaCf,FechaAlta)
		SELECT distinct beneficiarios.IdBenCP,
		case when SUBSTRING(PlanesFichas.[Plan],1,3)='NIV' THEN 'INI' ELSE SUBSTRING(PlanesFichas.[Plan],1,3) end CodCoberturaCf,		
		case when isnull(PlanesBeneficiarios.FechaAlta,GETDATE())<={ d '1900-01-01'} OR
				  isnull(PlanesBeneficiarios.FechaAlta,GETDATE())>{ d '2019-01-01'} THEN 
	          		GETDATE() ELSE isnull(PlanesBeneficiarios.FechaAlta,GETDATE()) END FechaAlta  
		
		FROM  gpieve.PieveSalud.dbo.PlanesFichas PlanesFichas 
		inner Join gpieve.PieveSalud.dbo.Beneficiarios Beneficiarios on PlanesFichas.IdTitularCf=Beneficiarios.IdTitularCf	
		AND planesfichas.IdRubroCf=2		--cobertura de salud
		and Beneficiarios.Existe=1
		Inner Join gpieve.PieveSalud.dbo.PlanesBeneficiarios PlanesBeneficiarios on PlanesBeneficiarios.IdRubroCf=PlanesFichas.idRubroCf and
		Beneficiarios.IdBenCp=PlanesBeneficiarios.IdBenCf 
		INNER JOIN gpieve.PieveSalud.dbo.Fichas f ON f.IdTitularCp=PlanesFichas.IdTitularCF AND f.Existe=1 AND f.IdTitularCp>0
		
		WHERE (f.IdTitularCp=@iIdTitularCp OR @iIdTitularCp=-1)
		AND case when SUBSTRING(PlanesFichas.[Plan],1,3)='NIV' THEN 'INI' ELSE SUBSTRING(PlanesFichas.[Plan],1,3) END Collate Database_Default
		IN (SELECT codcoberturacp FROM gpievenew.seodesarrollo2.dbo.Cobertura)
		
		
		
		
		SET @iIdError=@@ERROR
			
	END
	-- cobertura "PIE"

	if @iIdError=0
	BEGIN
		PRINT 'cobertura "PIE"'
		INSERT INTO gpievenew.seodesarrollo2.dbo.CoberturaBeneficiarios(IdBenCf,CodCoberturaCf,FechaAlta)	
		SELECT  DISTINCT cb.IdBenCf,'INI' CodCoberturaCf ,
		case when isnull(PlanesBeneficiarios.FechaAlta,GETDATE())<={ d '1900-01-01'} OR
				  isnull(PlanesBeneficiarios.FechaAlta,GETDATE())>{ d '2019-01-01'} THEN 
	          		GETDATE() ELSE isnull(PlanesBeneficiarios.FechaAlta,GETDATE()) END FechaAlta 
		FROM		
		gpieve.PieveSalud.dbo.PlanesFichas PlanesFichas 
		inner Join gpieve.PieveSalud.dbo.Beneficiarios Beneficiarios on PlanesFichas.IdTitularCf=Beneficiarios.IdTitularCf
